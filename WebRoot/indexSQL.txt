CREATE TABLE `index_contact` (
	`cNo` VARCHAR(50) NOT NULL,
	`cName` VARCHAR(30) NOT NULL,
	`cEMail` VARCHAR(30) NULL DEFAULT NULL,
	`cSubject` VARCHAR(50) NULL DEFAULT NULL,
	`cMessage` VARCHAR(500) NULL DEFAULT NULL,
	`cCurrentime` TIMESTAMP NULL DEFAULT NULL,
	`cStatus` VARCHAR(500) NULL DEFAULT NULL,
	`cIP` VARCHAR(50) NULL DEFAULT NULL,
	`cAddress` VARCHAR(100) NULL DEFAULT NULL,
	PRIMARY KEY (`cNo`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB;
